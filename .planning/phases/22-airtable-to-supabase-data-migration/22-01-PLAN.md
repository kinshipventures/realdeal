```sql
CREATE TYPE cadence AS ENUM ('weekly', 'biweekly', 'monthly', 'quarterly');
CREATE TYPE interaction_type AS ENUM ('call', 'email', 'text', 'meeting', 'intro', 'note', 'pod_change', 'field_update', 'categorization', 'pipeline_event', 'project_event', 'merge_event');
CREATE TYPE relationship_type AS ENUM ('Contact', 'Company');
CREATE TYPE relationship_status AS ENUM ('Active', 'Pending', 'Archived');
CREATE TYPE pipeline_status AS ENUM ('active', 'hidden');
CREATE TYPE opportunity_status AS ENUM ('open', 'won', 'lost', 'archived');
CREATE TYPE opportunity_priority AS ENUM ('high', 'medium', 'low');
CREATE TYPE campaign_type AS ENUM ('event', 'investment', 'outreach', 'other');
CREATE TYPE campaign_contact_status AS ENUM ('pending', 'reached', 'responded', 'confirmed');
CREATE TYPE campaign_status AS ENUM ('active', 'completed');
CREATE TYPE global_region AS ENUM ('AMER', 'APAC', 'ME', 'LATAM', 'EU');
CREATE TYPE gender_type AS ENUM ('Male', 'Female', 'Non-binary', 'Other');
CREATE TYPE contact_frequency AS ENUM ('Weekly', 'Monthly', 'Quarterly', 'Annual', 'As Needed');

```

**Tables in FK dependency order (per D-17):**

1. `pods` - columns: `id uuid DEFAULT gen_random_uuid() PRIMARY KEY`, `user_id uuid REFERENCES auth.users NOT NULL`, `name text NOT NULL`, `color text`, `owner owner_type`, `is_priority boolean DEFAULT false NOT NULL`, `cadence cadence`, `description text`, `capacity integer`, `enrichment_opt_in boolean DEFAULT false NOT NULL`, `created_at timestamptz DEFAULT now() NOT NULL`, `updated_at timestamptz DEFAULT now() NOT NULL`

2. `categories` - columns: `id uuid PK`, `user_id`, `pod_id uuid REFERENCES pods(id) ON DELETE CASCADE NOT NULL`, `name text NOT NULL`, `color text`, `created_at`, `updated_at`

3. `companies` - columns: `id uuid PK`, `user_id`, `name text NOT NULL`, `industry text`, `stage text`, `ticker text`, `domain text`, `website text`, `location text`, `notes text`, `custom_fields jsonb DEFAULT '{}'`, `created_at`, `updated_at`

4. `contacts` - columns: `id uuid PK`, `user_id`, `name text NOT NULL`, `first_name text`, `last_name text`, `email text`, `email_2 text`, `email_3 text`, `phone text`, `company text`, `company_id uuid REFERENCES companies(id) ON DELETE SET NULL`, `role text`, `location text`, `website text`, `linkedin text`, `country text`, `global_region global_region`, `gender gender_type`, `notes text`, `recommended_by text`, `introduced_by text`, `specialization text`, `past_clients text`, `birthday text` (kept as text - stores MM/DD or YYYY-MM-DD), `milestones text`, `interests text`, `relationship_context text`, `intel_notes text`, `relationship_owner text`, `contact_frequency contact_frequency`, `next_follow_up_date date`, `next_action text`, `kv_fund_investor text[]`, `spv_investor text[]`, `needs_review boolean DEFAULT false NOT NULL`, `last_contacted_at date`, `cadence_override cadence`, `type relationship_type DEFAULT 'Contact' NOT NULL`, `status relationship_status DEFAULT 'Active' NOT NULL`, `industry text`, `stage text`, `ticker text`, `domain text`, `custom_fields jsonb DEFAULT '{}'`, `created_at`, `updated_at`

Note: contacts has `industry`, `stage`, `ticker`, `domain` which overlap with companies. Per D-13, companies are split out. But contacts of type='Contact' still have these fields in the types.ts interface. Keep them on contacts too. This matches the existing types.ts interface.

5. `contact_pods` - junction (per D-08, D-14): `id uuid PK`, `user_id`, `contact_id uuid REFERENCES contacts(id) ON DELETE CASCADE NOT NULL`, `pod_id uuid REFERENCES pods(id) ON DELETE CASCADE NOT NULL`, `is_primary boolean DEFAULT false NOT NULL`, `created_at`, UNIQUE(contact_id, pod_id)

6. `contact_categories` - junction (per D-08): `id uuid PK`, `user_id`, `contact_id uuid REFERENCES contacts(id) ON DELETE CASCADE NOT NULL`, `category_id uuid REFERENCES categories(id) ON DELETE CASCADE NOT NULL`, `created_at`, UNIQUE(contact_id, category_id)

7. `interactions` - columns: `id uuid PK`, `user_id`, `contact_id uuid REFERENCES contacts(id) ON DELETE CASCADE NOT NULL`, `type interaction_type NOT NULL`, `date date NOT NULL`, `notes text`, `summary text`, `source interaction_source`, `email_link text`, `granola_link text`, `event_detail text`, `actor text`, `created_at`, `updated_at`

8. `pipelines` - columns: `id uuid PK`, `user_id`, `name text NOT NULL`, `status pipeline_status DEFAULT 'active' NOT NULL`, `created_at`, `updated_at`

9. `pipeline_stages` - columns: `id uuid PK`, `user_id`, `pipeline_id uuid REFERENCES pipelines(id) ON DELETE CASCADE NOT NULL`, `name text NOT NULL`, `color text`, `"order" integer NOT NULL` (order is a reserved word - must be quoted), `created_at`, `updated_at`

10. `opportunities` - columns: `id uuid PK`, `user_id`, `name text NOT NULL`, `stage_id uuid REFERENCES pipeline_stages(id) ON DELETE SET NULL`, `notes text`, `priority opportunity_priority`, `status opportunity_status DEFAULT 'open' NOT NULL`, `created_at`, `updated_at`

11. `opportunity_contacts` - junction (per D-08): `id uuid PK`, `user_id`, `opportunity_id uuid REFERENCES opportunities(id) ON DELETE CASCADE NOT NULL`, `contact_id uuid REFERENCES contacts(id) ON DELETE CASCADE NOT NULL`, `created_at`, UNIQUE(opportunity_id, contact_id)

12. `campaigns` - columns: `id uuid PK`, `user_id`, `name text NOT NULL`, `type campaign_type NOT NULL`, `deadline date`, `status campaign_status DEFAULT 'active' NOT NULL`, `created_at`, `updated_at`

13. `campaign_contacts` - junction: `id uuid PK`, `user_id`, `campaign_id uuid REFERENCES campaigns(id) ON DELETE CASCADE NOT NULL`, `contact_id uuid REFERENCES contacts(id) ON DELETE CASCADE NOT NULL`, `status campaign_contact_status DEFAULT 'pending' NOT NULL`, `notes text`, `created_at`, `updated_at`, UNIQUE(campaign_id, contact_id)

14. `projects` - columns: `id uuid PK`, `user_id`, `name text NOT NULL`, `description text`, `owner text`, `notes text`, `created_at`, `updated_at`

15. `project_contacts` - junction: `id uuid PK`, `user_id`, `project_id uuid REFERENCES projects(id) ON DELETE CASCADE NOT NULL`, `contact_id uuid REFERENCES contacts(id) ON DELETE CASCADE NOT NULL`, `created_at`, UNIQUE(project_id, contact_id)

16. `project_opportunities` - junction: `id uuid PK`, `user_id`, `project_id uuid REFERENCES projects(id) ON DELETE CASCADE NOT NULL`, `opportunity_id uuid REFERENCES opportunities(id) ON DELETE CASCADE NOT NULL`, `created_at`, UNIQUE(project_id, opportunity_id)

17. `field_config` - columns: `id uuid PK`, `user_id`, `name text NOT NULL`, `airtable_field_id text`, `field_type text NOT NULL`, `scope_type text NOT NULL`, `scope_pod_id uuid REFERENCES pods(id) ON DELETE SET NULL`, `required boolean DEFAULT false NOT NULL`, `display_order integer DEFAULT 0 NOT NULL`, `created_at`, `updated_at`

18. `_migration_id_map` - temp table (per D-16): `id uuid PK`, `user_id`, `airtable_id text NOT NULL`, `table_name text NOT NULL`, `supabase_uuid uuid NOT NULL`, `created_at`, UNIQUE(airtable_id, table_name)

**RLS policies (per D-19, D-20):**
For every table: `ALTER TABLE {table} ENABLE ROW LEVEL SECURITY;` then `CREATE POLICY "{table}_owner" ON {table} FOR ALL USING (user_id = auth.uid());`

**updated_at trigger (per Pitfall 6 from RESEARCH.md):**
Create a function `update_updated_at()` that sets `NEW.updated_at = now()` and apply it as a BEFORE UPDATE trigger on every table that has `updated_at`.

**Index recommendations (include as comments):**
- `contacts(user_id)` - RLS performance
- `contacts(last_contacted_at)` - overdue queries
- `interactions(contact_id)` - interaction history lookups
- `contact_pods(contact_id)`, `contact_pods(pod_id)` - join performance
- `contact_categories(contact_id)`, `contact_categories(category_id)` - join performance
- `pipeline_stages(pipeline_id)` - stage ordering
